CREATE DATABASE VIDLY;

USE VIDLY;

CREATE TABLE Users
(uid INT NOT NULL AUTO_INCREMENT,
email VARCHAR(64) NOT NULL UNIQUE,
password VARCHAR(1024) NOT NULL,
firstname VARCHAR(64) NOT NULL,
lastname VARCHAR(64) NOT NULL,
birthdate DATE,
gender CHAR(1),
phone_number VARCHAR(32),
address VARCHAR(256),
created_at DATE,
PRIMARY KEY(uid));
